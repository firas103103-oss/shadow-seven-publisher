detectProjectType(files) {\n  if (files.includes('package.json')) {\n    if (files.includes('tsconfig.json')) return 'TypeScript';\n    if (files.includes('vite.config.js')) return 'Vite';\n    if (files.includes('next.config.js')) return 'Next.js';\n    if (files.includes('expo.config.js')) return 'React Native';\n    return 'Node.js';\n  }\n  if (files.includes('requirements.txt')) return 'Python';\n  if (files.includes('go.mod')) return 'Go';\n  if (files.includes('pom.xml')) return 'Java';\n  if (files.includes('Dockerfile')) return 'Containerized';\n  return type: 'database', priority: 'critical' });\n  }\n\n  // Check authentication\n  if (!projectFiles.includes('AuthPage.jsx')) {\n    gaps.push({ type: 'auth', priority: 'critical' });\n  }\n\n  // Check exports\n  if (!projectFiles.includes('exportFunctions.js')) {\n    gaps.push({ type: 'export', priority: 'high' });\n  }\n\n  // Check tests\n  const testFiles = projectFiles.filter(f => f.includes('.test.'));\n  if (testFiles.length === 0) {\n    gaps.push({ type: 'testing', priority: 'high' });\n  }\n\n  return gaps.sort((a, b) => {\n    const priority = { critical: 0, high: 1, medium: 2, low: 3 };\n    return priority[a.priority] - priority[b.priority];\n  });\n}\n```\n\n### Completion Calculation\n\n```javascript\nfunction calculateCompletion(audit) {\n  const components = 0);\n\n  return {\n    percentage: (completed / total) * 100,\n    completed,\n    total,\n    components\n  };\n}\n```\n\n---\n\n## File Templates\n\n### Database Schema Template\n```sql\n-- Users
